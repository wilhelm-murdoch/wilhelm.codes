{{/* Fetch a GitHub API collection as a slice. Params: "url" (endpoint), "fixture" (committed JSON fallback path). Always fetches live, authenticating with HUGO_GITHUB_TOKEN when set: the unauthenticated API is 60 req/hr per IP, which shared CI build IPs blow through (a 403); a token raises it to 5000/hr. Any failure (offline, rate limit, outage) falls back to the fixture so a build never dies on GitHub. Hugo caches responses, so dev rebuilds don't re-hit the API. */}} {{- $url := .url -}} {{- $fixture := .fixture -}} {{- $data := slice -}} {{- $opts := dict -}} {{- with os.Getenv "HUGO_GITHUB_TOKEN" -}} {{- $opts = dict "headers" (dict "Authorization" (printf "Bearer %s" .)) -}} {{- end -}} {{- $remote := try (resources.GetRemote $url $opts) -}} {{- with $remote.Value -}} {{- $data = . | transform.Unmarshal -}} {{- else -}} {{- with $remote.Err -}} {{- warnf "changelog: live fetch failed for %s (%s); using %s" $url . $fixture -}} {{- end -}} {{- /* Tolerate an empty/unparseable fixture (e.g. a repo with no releases). */ -}} {{- $parsed := try (transform.Unmarshal (os.ReadFile $fixture)) -}} {{- with $parsed.Value -}}{{- $data = . -}}{{- end -}} {{- end -}} {{- return $data -}}
